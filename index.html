const express = require("express"); const app = express(); const path = require("path"); app.use(express.json()); app.use(express.static("public")); let votes = { optionA: 0, optionB: 0, }; let votedIPs = new Set(); // pega IP do usuário function getIP(req) { return req.headers["x-forwarded-for"] || req.socket.remoteAddress; } // rota de voto app.post("/vote", (req, res) => { const ip = getIP(req); const { option } = req.body; if (votedIPs.has(ip)) { return res.status(403).json({ message: "Você já votou!" }); } if (!votes.hasOwnProperty(option)) { return res.status(400).json({ message: "Opção inválida" }); } votes[option] += 1; votedIPs.add(ip); return res.json({ message: "Voto registrado com sucesso!", votes, }); }); // pegar resultado app.get("/results", (req, res) => { res.json(votes); }); app.listen(3000, () => { console.log("uf.7 rodando em http://localhost:3000"); });
