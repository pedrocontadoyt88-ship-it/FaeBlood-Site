<script>
let votes = JSON.parse(localStorage.getItem("votes")) || {};
let voted = localStorage.getItem("voted");

let candidates = JSON.parse(localStorage.getItem("candidates")) || [
    "Candidato A",
    "Candidato B",
    "Candidato C"
];

function initVotes(){
    candidates.forEach(c=>{
        if(votes[c] === undefined) votes[c]=0;
    });
    localStorage.setItem("votes", JSON.stringify(votes));
}
initVotes();

function render(){
    let container = document.getElementById("container");
    container.innerHTML = "";

    candidates.forEach(c=>{
        let div = document.createElement("div");
        div.className="card";
        div.innerHTML = `<h3>${c}</h3>
        <button class='vote' onclick="vote('${c}')">Votar</button>`;
        container.appendChild(div);
    });
}
render();

function vote(cand){
    if(voted){
        document.getElementById("msg").innerText="❌ Você já votou!";
        return;
    }

    votes[cand]++;
    localStorage.setItem("votes", JSON.stringify(votes));
    localStorage.setItem("voted", "true");

    document.getElementById("msg").innerText="✅ Voto registrado!";
}
